-- 1. Table: adminlogin_tb
CREATE TABLE `adminlogin_tb` (
  `a_login_id` int(11) NOT NULL AUTO_INCREMENT,
  `a_name` varchar(30) NOT NULL,
  `a_email` varchar(30) NOT NULL,
  `a_password` varchar(20) NOT NULL,
  `a_theme` enum('light','dark') DEFAULT 'light',
  `a_created_at` timestamp NOT NULL DEFAULT current_timestamp(),
  `a_otp` varchar(8) DEFAULT NULL,
  `a_otp_expiry` datetime(6) DEFAULT NULL,
  `a_verified` tinyint(1) NOT NULL DEFAULT 0,
  PRIMARY KEY (`a_login_id`)
) ENGINE=MyISAM DEFAULT CHARSET=latin1;
